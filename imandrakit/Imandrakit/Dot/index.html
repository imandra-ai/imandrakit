Module Imandrakit.Dot

Pretty printing in the DOT (graphviz) format.

Example (print divisors from 42):

  let open Dot in
  with_out "/tmp/truc.dot"
    (fun out ->
       pp ~attrs_v:(fun i -> [`Label (string_of_int i)]) ~graph:divisors_graph out 42
    )
